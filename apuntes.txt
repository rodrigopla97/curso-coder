BLOQUE
{
  var primera = "variable var";
  let segunda = "variable let";

  console.log(primera);
  console.log(segunda);
  // Se muestran en consola
}
console.log(primera);
console.log(segunda);
// Solo se muestra var ya que "let" es variable local (de bloque) y "var" global
